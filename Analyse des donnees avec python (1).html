def sumsq(x,y):
    return sum((x - y)**2)

def r2score(pred , target):
    return 1 - sumsq(pred, target) / sumsq(target, np.mean(target))

score_bl = r2score(baseline_pred[:,0],ytest["Price"])

print(score_bl)

    ytest_pss = ytest[ts_pss]
